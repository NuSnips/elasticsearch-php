$params = [
    'body' => [
        'size' => 0,
        'runtime_mappings' => [
            'normalized_genre' => [
                'type' => 'keyword',
                'script' => '
        String genre = doc[\'genre\'].value;
        if (doc[\'product\'].value.startsWith(\'Anthology\')) {
          emit(genre + \' anthology\');
        } else {
          emit(genre);
        }
      ',
            ],
        ],
        'aggs' => [
            'genres' => [
                'terms' => [
                    'field' => 'normalized_genre',
                ],
            ],
        ],
    ],
];
$response = $client->search($params);
